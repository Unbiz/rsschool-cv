function squareDigits(num) {
  const sum = String(num)
    .split('')
    .map((item) => Number(item) ** 2)
    .join('');
  return Number(sum);
}

function findShort(s) {
  const arrOfLengths = s
    .split(' ')
    .map((word) => word.length);
  return Math.min(...arrOfLengths);
}
